  <script>

    // Create map
    var map = L.map('map').setView([42.3912, -72.5267], 13);

    // Basemap layers
    var topoMap = L.tileLayer('https://{s}.tile.opentopomap.org/{z}/{x}/{y}.png', {
      maxZoom: 17,
      attribution: '© OpenStreetMap contributors | OpenTopoMap'
    }).addTo(map);

    var streetMap = L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      maxZoom: 19,
      attribution: '© OpenStreetMap contributors'
    });

    // Layer groups
    var campusLayer = L.layerGroup().addTo(map);
    var parkLayer = L.layerGroup().addTo(map);
    var trailLayer = L.layerGroup().addTo(map);
    var waterLayer = L.layerGroup().addTo(map);

    // Function for points
    function addPoint(lat, lng, title, type, description, group, color) {

      L.circleMarker([lat, lng], {
        radius: 8,
        color: "white",
        weight: 2,
        fillColor: color,
        fillOpacity: 0.9

      }).addTo(group)

      .bindPopup(
        "<div class='popup-title'>" + title + "</div>" +
        "<div class='popup-type'>" + type + "</div>" +
        "<div>" + description + "</div>"
      );
    }

    // Campus points
    addPoint(
      42.3912, -72.5267,
      "UMass Amherst",
      "Campus / Starting Point",
      "A central starting point for exploring nearby outdoor spaces around Amherst.",
      campusLayer,
      "#2b7bff"
    );

    addPoint(
      42.3894, -72.5283,
      "Campus Pond Area",
      "Campus Outdoor Space",
      "A familiar UMass location for short walks and outdoor recreation.",
      campusLayer,
      "#2b7bff"
    );

    // Parks
    addPoint(
      42.3736, -72.5198,
      "Groff Park",
      "Park / Open Space",
      "A recreational park with open space and nearby walking access.",
      parkLayer,
      "#2ca25f"
    );

    addPoint(
      42.3765, -72.5195,
      "Kendrick Park",
      "Downtown Green Space",
      "A small green space close to downtown Amherst.",
      parkLayer,
      "#2ca25f"
    );

    addPoint(
      42.3858, -72.5321,
      "Mill River Recreation Area",
      "Park / Recreation Area",
      "A larger recreation area with trails and open fields.",
      parkLayer,
      "#2ca25f"
    );

    addPoint(
      42.4038, -72.5314,
      "Mill River Conservation Area",
      "Conservation Area",
      "A quieter wooded natural area near North Amherst.",
      parkLayer,
      "#2ca25f"
    );

    // Trails
    addPoint(
      42.3976, -72.5034,
      "Amethyst Brook Conservation Area",
      "Trail / Conservation Area",
      "A wooded trail area for hiking and outdoor recreation.",
      trailLayer,
      "#8856a7"
    );

    addPoint(
      42.3759, -72.5130,
      "Norwottuck Rail Trail Access",
      "Trail Access",
      "A paved multi-use trail for walking and biking.",
      trailLayer,
      "#8856a7"
    );

    addPoint(
      42.3842, -72.4917,
      "Hampshire College Trails",
      "Trail / Natural Area",
      "A nearby outdoor area with trails and open fields.",
      trailLayer,
      "#8856a7"
    );

    addPoint(
      42.3988, -72.4756,
      "Mount Orient / Pelham Hills Area",
      "Natural Area",
      "A wooded scenic area east of Amherst.",
      trailLayer,
      "#8856a7"
    );

    // Water areas
    addPoint(
      42.4122, -72.5304,
      "Puffers Pond",
      "Waterfront Area",
      "A popular pond and walking area in North Amherst.",
      waterLayer,
      "#2b8cbe"
    );

    addPoint(
      42.4083, -72.5171,
      "Cushman Brook Area",
      "Water / Natural Area",
      "A stream corridor and wooded landscape area.",
      waterLayer,
      "#2b8cbe"
    );

    // Layer controls
    var baseMaps = {
      "Topographic Map": topoMap,
      "Street Map": streetMap
    };

    var overlayMaps = {
      "Campus Points": campusLayer,
      "Parks and Open Spaces": parkLayer,
      "Trails and Conservation Areas": trailLayer,
      "Waterfront Areas": waterLayer
    };

    L.control.layers(baseMaps, overlayMaps, {
      collapsed: false
    }).addTo(map);

    // Scale bar
    L.control.scale({
      imperial: true,
      metric: true
    }).addTo(map);

  </script>
